#include <iostream>

using namespace std;                // 3 variant

int main()
{
    int n;
    cout << "array length [0;100]\n";
    cin >> n;
    while (n < 0 || n > 100)
    {
        cout << "Wrong data\n";
        cin >> n;
    }
    int* mas=new int[n];
    cout << "Input array\n";
    for (int i = 0; i < n; i++) {
        cin >> mas[i];
    }
    if (n <=1)
    {
        cout << "empty";
        delete[]mas;
        return 0;
    }
    for (int i = 1; i < n - 1; i++) {
        if (mas[i]/mas[i-1]!=mas[i+1]/mas[i])
        {
            cout << "empty";
            delete[]mas;
            return 0;
        }
    }
    cout << float(mas[1]) / mas[0]; 
    delete[]mas;
}
